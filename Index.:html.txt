</div>

<script>
let board = Array(9).fill("");
let player = "O";
let ai = "X";
let gameOver = false;

function startGame(){
    document.getElementById("home").style.display="none";
    document.getElementById("game").style.display="block";
    resetBoard();
}

function resetBoard(){
    board = Array(9).fill("");
    gameOver=false;
    const boardDiv=document.getElementById("board");
    boardDiv.innerHTML="";
    for(let i=0;i<9;i++){
        const cell=document.createElement("div");
        cell.className="cell";
        cell.onclick=()=>playerMove(i);
        boardDiv.appendChild(cell);
    }
    document.getElementById("status").textContent="Your turn!";
}

function playerMove(index){
    if(gameOver || board[index]!=="") return;
    board[index]=player;
    render();
    if(checkWinner(board,player)){
        endGame("You Win!");
        return;
    }
    if(board.every(c=>c!=="")){
        endGame("It's a Draw!");
        return;
    }
    document.getElementById("status").textContent="AI is thinking...";
    setTimeout(aiMove,500);
}

function aiMove(){
    let empty = board.map((v,i)=>v===""?i:null).filter(v=>v!==null);
    let move = empty[Math.floor(Math.random()*empty.length)];
    board[move]=ai;
    render();
    if(checkWinner(board,ai)){
        endGame("AI Wins!");
        return;
    }
    if(board.every(c=>c!=="")){
        endGame("It's a Draw!");
        return;
    }
    document.getElementById("status").textContent="Your turn!";
}

function render(){
    const cells=document.querySelectorAll(".cell");
    board.forEach((v,i)=>{
        cells[i].textContent=v;
        cells[i].className = "cell"+(v===""?"":(" "+v.toLowerCase()));
    });
}

function checkWinner(b,player){
    const wins=[
        [0,1,2],[3,4,5],[6,7,8],
        [0,3,6],[1,4,7],[2,5,8],
        [0,4,8],[2,4,6]
    ];
    return wins.some(line=>line.every(i=>b[i]===player));
}

function endGame(msg){
    gameOver=true;
    document.getElementById("status").textContent=msg+" Tap any cell to restart.";
    setTimeout(()=>{resetBoard();},2000);
}
